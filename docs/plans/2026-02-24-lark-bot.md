```typescript
import type { ChatProvider, ChatEvent } from "./types.js";
import type { SessionManager } from "../sessions/manager.js";
import type { EventLog } from "../core/event-bus.js";
import type { MemoryManager } from "../memory/manager.js";
import type { MemoryFlushFn } from "./compaction.js";
import { compactHistory } from "./compaction.js";
import { extractMemories } from "../memory/extractor.js";
import { createHubEvent } from "../core/hub-event.js";

export type ConversationDeps = {
  provider: ChatProvider;
  sessionManager: SessionManager;
  eventLog: EventLog;
  memoryManager?: MemoryManager;
  maxHistoryMessages?: number;
  maxHistoryTokens?: number;
};

export type ConversationRequest = {
  userId: string;
  message: string;
  sessionId?: string;
  channel: string;
  channelId: string;
  deps: ConversationDeps;
  abortSignal?: AbortSignal;
};

export type ConversationResult = {
  text: string;
  sessionId: string;
  costUsd: number;
  events: ChatEvent[];
  error?: string;
};

// Per-session concurrency lock
const sessionLocks = new Map<string, { promise: Promise<void>; count: number }>();

function withSessionLock<T>(
  sessionId: string,
  fn: () => Promise<T>,
): Promise<T> {
  const entry = sessionLocks.get(sessionId);
  const prev = entry?.promise ?? Promise.resolve();
  const count = (entry?.count ?? 0) + 1;
  let unlock: () => void;
  const next = new Promise<void>((r) => (unlock = r));
  sessionLocks.set(sessionId, { promise: next, count });
  return prev.then(() => fn()).finally(() => {
    unlock();
    const current = sessionLocks.get(sessionId);
    if (current) {
      current.count--;
      if (current.count <= 0) {
        sessionLocks.delete(sessionId);
      }
    }
  });
}

/**
 * Core conversation handler — shared by Web and Lark channels.
 *
 * Flow: resolve session → record user message → load history → search memories →
 * compact → inject identity → stream provider → collect reply → save → audit
 */
export async function handleConversation(
  req: ConversationRequest,
): Promise<ConversationResult> {
  const { userId, message, channel, channelId, deps, abortSignal } = req;
  const {
    provider,
    sessionManager,
    eventLog,
    memoryManager,
    maxHistoryMessages = 40,
    maxHistoryTokens = 0,
  } = deps;

  // 1. Resolve or create session
  let session = req.sessionId ? sessionManager.getById(req.sessionId) : null;

  // Don't reuse a session from a different user
  if (session && session.userId !== userId) {
    session = null;
  }

  if (!session) {
    session = sessionManager.create({
      userId,
      channel,
      channelId,
      provider: provider.name,
    });
  }

  // 2. Per-session lock
  return withSessionLock(session.id, async () => {
    const t0 = Date.now();

    // 3. Record user message
    sessionManager.appendMessage(session.id, { role: "user", content: message });

    // 4. Load and compact history
    const rawHistory = sessionManager.getMessages(session.id).map((m) => ({
      role: m.role,
      content: m.content,
    }));

    // 4a. Search relevant memories
    let memoryText: string | null = null;
    if (memoryManager) {
      try {
        const memories = memoryManager.search(session.userId, message);
        if (memories.length > 0) {
          memoryText = memories
            .map((m) => `- id=${m.id} [${m.category}] ${m.key}: ${m.value}`)
            .join("\n");
        }
      } catch { /* best-effort */ }
    }

    // 4b. Build memoryFlush callback
    const memoryFlush: MemoryFlushFn | undefined =
      memoryManager && provider.summarize
        ? async (early) => {
            const existing = memoryManager.getByUser(session.userId);
            const items = await extractMemories(
              early,
              (prompt) => provider.summarize!([{ role: "user", content: prompt }]),
              existing,
            );
            if (items.length > 0) {
              memoryManager.save(session.userId, items, session.id);
            }
          }
        : undefined;

    const history = await compactHistory(rawHistory, {
      maxMessages: maxHistoryMessages,
      maxTokens: maxHistoryTokens,
      summarize: provider.summarize?.bind(provider),
      memoryFlush,
    });

    // Persist compacted history
    if (history.length < rawHistory.length) {
      const keepCount = history.length - 1;
      sessionManager.compactMessages(session.id, keepCount, history[0]);
    }

    // 4c. Inject user identity + memories (not persisted)
    const identityParts: string[] = [];
    identityParts.push(`当前用户: ${session.userId}`);
    if (memoryText) {
      identityParts.push(`你了解该用户的以下信息:\n${memoryText}`);
    }
    history.unshift({ role: "system", content: identityParts.join("\n\n") });

    // 5. Stream from provider and collect results
    let assistantText = "";
    let costUsd = 0;
    const events: ChatEvent[] = [];
    let error: string | undefined;

    try {
      for await (const event of provider.stream({
        message,
        sessionId: session.providerSessionId ?? undefined,
        history,
        abortSignal,
        toolContext: { userId: session.userId, sessionId: session.id },
      })) {
        if (event.type === "text") {
          assistantText += event.content;
        }
        if (event.type === "done") {
          costUsd = event.costUsd;
          // Save assistant reply
          if (assistantText) {
            sessionManager.appendMessage(session.id, {
              role: "assistant",
              content: assistantText,
            });
          }
          // Store provider session ID
          if (event.sessionId) {
            sessionManager.updateProviderSessionId(session.id, event.sessionId);
          }
          // Rewrite sessionId to ours
          events.push({ ...event, sessionId: session.id });
        } else {
          events.push(event);
        }
      }
    } catch (err: any) {
      error = err.message ?? String(err);
      // Save partial response
      if (assistantText) {
        sessionManager.appendMessage(session.id, {
          role: "assistant",
          content: assistantText,
        });
      }
    }

    // 6. Audit log
    try {
      eventLog.log(createHubEvent({
        type: `chat.${channel}`,
        source: `${channel}_chat`,
        payload: { message, costUsd },
        context: { sessionId: session.id, userId },
      }));
    } catch { /* best-effort */ }

    return {
      text: assistantText,
      sessionId: session.id,
      costUsd,
      events,
      error,
    };
  });
}
```

**Step 1: Refactor `chatRouter` to delegate to `handleConversation()`**

Replace the core logic (lines 120-296) with a call to `handleConversation()`, keeping the SSE streaming wrapper and slash command handling in the router. The router becomes a thin Web-channel adapter.

```typescript
import type { Hono } from "hono";
import { streamSSE } from "hono/streaming";
import type { ChatProvider } from "./types.js";
import type { SessionManager } from "../sessions/manager.js";
import type { EventLog } from "../core/event-bus.js";
import type { MemoryManager } from "../memory/manager.js";
import { handleCommand } from "./commands.js";
import { handleConversation, type ConversationDeps } from "./conversation.js";

type ChatRouterDeps = {
  sessionManager: SessionManager;
  eventLog: EventLog;
  maxHistoryMessages?: number;
  maxHistoryTokens?: number;
  memoryManager?: MemoryManager;
  skillsDirs: string[];
};

export function chatRouter(
  app: Hono,
  provider: ChatProvider,
  deps: ChatRouterDeps,
): void {
  const { sessionManager, eventLog, memoryManager } = deps;

  const convDeps: ConversationDeps = {
    provider,
    sessionManager,
    eventLog,
    memoryManager,
    maxHistoryMessages: deps.maxHistoryMessages ?? 40,
    maxHistoryTokens: deps.maxHistoryTokens ?? 0,
  };

  // Lightweight auth check
  app.get("/api/chat", (c) => {
    const userId = c.get("userId") ?? "web-anonymous";
    return c.json({ status: "ok", userId });
  });

  app.post("/api/chat", async (c) => {
    const body = await c.req.json().catch(() => ({}));
    const message = body.message;
    const sessionId: string | undefined = body.sessionId;

    if (!message || typeof message !== "string") {
      return c.json({ error: "message is required" }, 400);
    }

    const userId = c.get("userId") ?? "web-anonymous";

    // Handle slash commands (before LLM call)
    let session = sessionId ? sessionManager.getById(sessionId) : null;
    if (session && session.userId !== userId) session = null;

    if (!session) {
      session = sessionManager.create({
        userId,
        channel: "web",
        channelId: "",
        provider: provider.name,
      });
    }

    const installDir = deps.skillsDirs[1] ?? deps.skillsDirs[0];
    const cmdResult = await handleCommand(message, {
      session,
      sessionManager,
      providerName: provider.name,
      installDir,
      skillsDirs: deps.skillsDirs,
    });
    if (cmdResult) {
      if (cmdResult.newSession) session = cmdResult.newSession;
      return streamSSE(c, async (stream) => {
        for (const event of cmdResult.events) {
          await stream.writeSSE({ data: JSON.stringify(event) });
        }
      });
    }

    // Delegate to shared conversation handler, stream results as SSE
    return streamSSE(c, async (stream) => {
      const heartbeat = setInterval(async () => {
        try { await stream.writeSSE({ data: "", event: "heartbeat" }); } catch {}
      }, 15_000);

      try {
        const result = await handleConversation({
          userId,
          message,
          sessionId: session.id,
          channel: "web",
          channelId: "",
          deps: convDeps,
          abortSignal: c.req.raw.signal,
        });

        for (const event of result.events) {
          await stream.writeSSE({ data: JSON.stringify(event) });
        }

        if (result.error) {
          await stream.writeSSE({
            data: JSON.stringify({ type: "error", message: result.error }),
          });
          await stream.writeSSE({
            data: JSON.stringify({ type: "done", sessionId: result.sessionId, costUsd: result.costUsd }),
          });
        }
      } catch (err: any) {
        try {
          await stream.writeSSE({
            data: JSON.stringify({ type: "error", message: err.message ?? "unknown error" }),
          });
        } catch {}
      } finally {
        clearInterval(heartbeat);
      }
    });
  });
}
```

**Important note:** The refactored router loses real-time SSE streaming of individual `text` chunks — all events are collected first then sent. This changes the UX for web chat. If real-time streaming is essential, an alternative approach is to make `handleConversation` accept an optional `onEvent` callback that the web router calls for real-time SSE, while the Lark handler ignores. Evaluate during implementation whether this matters for the current use case. If it does, add `onEvent?: (event: ChatEvent) => void` to `ConversationRequest` and call it in the stream loop, then have the web router use it for real-time SSE. The Lark handler can either ignore it or use it for periodic card updates.

```typescript
import * as lark from "@larksuiteoapi/node-sdk";
import type { Env } from "../env.js";

export type LarkClient = lark.Client;

export function createLarkClient(env: Env): LarkClient {
  if (!env.LARK_APP_ID || !env.LARK_APP_SECRET) {
    throw new Error("LARK_APP_ID and LARK_APP_SECRET are required");
  }

  return new lark.Client({
    appId: env.LARK_APP_ID,
    appSecret: env.LARK_APP_SECRET,
    appType: lark.AppType.SelfBuild,
    domain: lark.Domain.Feishu,
  });
}

/** Build a markdown card JSON for interactive messages. */
export function buildMarkdownCard(text: string): string {
  return JSON.stringify({
    schema: "2.0",
    config: { wide_screen_mode: true },
    body: {
      elements: [{ tag: "markdown", content: text }],
    },
  });
}

/** Send a new interactive card message to a chat. Returns the message_id. */
export async function sendCard(
  client: LarkClient,
  chatId: string,
  markdown: string,
): Promise<string> {
  const res = await client.im.message.create({
    params: { receive_id_type: "chat_id" },
    data: {
      receive_id: chatId,
      content: buildMarkdownCard(markdown),
      msg_type: "interactive",
    },
  });
  return res.data?.message_id ?? "";
}

/** Update an existing interactive card message. */
export async function patchCard(
  client: LarkClient,
  messageId: string,
  markdown: string,
): Promise<void> {
  await client.im.message.patch({
    path: { message_id: messageId },
    data: { content: buildMarkdownCard(markdown) },
  });
}
```

```typescript
import type { Hono } from "hono";
import type { ChatProvider } from "../chat/types.js";
import type { SessionManager } from "../sessions/manager.js";
import type { EventLog } from "../core/event-bus.js";
import type { MemoryManager } from "../memory/manager.js";
import { handleConversation, type ConversationDeps } from "../chat/conversation.js";

export type LarkRouterDeps = {
  provider: ChatProvider;
  sessionManager: SessionManager;
  eventLog: EventLog;
  memoryManager?: MemoryManager;
  maxHistoryMessages?: number;
  maxHistoryTokens?: number;
  sendCard: (chatId: string, markdown: string) => Promise<string>;
  patchCard: (messageId: string, markdown: string) => Promise<void>;
  verificationToken?: string;
};

// Simple dedup cache — TTL 5 minutes
const processedMessages = new Map<string, number>();
const DEDUP_TTL_MS = 5 * 60 * 1000;

function isDuplicate(messageId: string): boolean {
  const now = Date.now();
  // Cleanup old entries
  for (const [id, ts] of processedMessages) {
    if (now - ts > DEDUP_TTL_MS) processedMessages.delete(id);
  }
  if (processedMessages.has(messageId)) return true;
  processedMessages.set(messageId, now);
  return false;
}

export function larkRouter(app: Hono, deps: LarkRouterDeps): void {
  const {
    provider,
    sessionManager,
    eventLog,
    memoryManager,
    sendCard,
    patchCard,
  } = deps;

  const convDeps: ConversationDeps = {
    provider,
    sessionManager,
    eventLog,
    memoryManager,
    maxHistoryMessages: deps.maxHistoryMessages ?? 40,
    maxHistoryTokens: deps.maxHistoryTokens ?? 0,
  };

  app.post("/api/lark/webhook", async (c) => {
    const body = await c.req.json().catch(() => ({}));

    // 1. URL verification challenge
    if (body.type === "url_verification") {
      return c.json({ challenge: body.challenge });
    }

    // 2. Extract event data
    const event = body.event;
    if (!event?.message) {
      return c.json({ code: 0 });
    }

    const msg = event.message;
    const messageId: string = msg.message_id ?? "";
    const chatId: string = msg.chat_id ?? "";
    const chatType: string = msg.chat_type ?? "p2p";
    const msgType: string = msg.message_type ?? "";
    const senderOpenId: string = event.sender?.sender_id?.open_id ?? "";

    // 3. Only handle text messages
    if (msgType !== "text") {
      return c.json({ code: 0 });
    }

    // 4. Dedup
    if (isDuplicate(messageId)) {
      return c.json({ code: 0 });
    }

    // 5. Parse text content
    let text: string;
    try {
      const content = JSON.parse(msg.content);
      text = content.text ?? "";
    } catch {
      return c.json({ code: 0 });
    }

    if (!text.trim()) {
      return c.json({ code: 0 });
    }

    // 6. Respond immediately (Lark requires < 3s response), process async
    const userId = `lark:${senderOpenId}`;

    // Fire-and-forget: process message asynchronously
    processLarkMessage({
      userId,
      text: text.trim(),
      chatId,
      chatType,
      convDeps,
      sessionManager,
      sendCard,
      patchCard,
    }).catch((err) => {
      console.error("[lark] async processing error:", err);
    });

    return c.json({ code: 0 });
  });
}

async function processLarkMessage(opts: {
  userId: string;
  text: string;
  chatId: string;
  chatType: string;
  convDeps: ConversationDeps;
  sessionManager: SessionManager;
  sendCard: (chatId: string, markdown: string) => Promise<string>;
  patchCard: (messageId: string, markdown: string) => Promise<void>;
}): Promise<void> {
  const { userId, text, chatId, convDeps, sessionManager, sendCard, patchCard } = opts;

  // Send "thinking" card
  let replyMsgId: string;
  try {
    replyMsgId = await sendCard(chatId, "思考中...");
  } catch (err) {
    console.error("[lark] failed to send thinking card:", err);
    return;
  }

  // Find or create session — reuse active session for same user+chat
  const existingSession = sessionManager.findActive(userId, "lark");
  const sessionId =
    existingSession && existingSession.channelId === chatId
      ? existingSession.id
      : undefined;

  // Run conversation
  const result = await handleConversation({
    userId,
    message: text,
    sessionId,
    channel: "lark",
    channelId: chatId,
    deps: convDeps,
  });

  // Patch card with final reply
  const replyText = result.error
    ? `Error: ${result.error}\n\n${result.text || ""}`
    : result.text || "(empty response)";

  try {
    await patchCard(replyMsgId, replyText);
  } catch (err) {
    console.error("[lark] failed to patch reply card:", err);
  }
}
```

**Critical design decisions:**
- `handleConversation()` collects all events then returns — not real-time streaming. For Lark this is correct (send complete reply). For Web, this means losing SSE streaming of individual text chunks. Task 3 includes a note about adding an `onEvent` callback if real-time streaming is needed.
- Lark replies use "send card → patch card" pattern for smooth UX ("思考中..." → final answer).
- Fire-and-forget async processing because Lark requires webhook response < 3 seconds.
- Message dedup by `message_id` with 5-minute TTL (Lark may resend events).
- `sendCard`/`patchCard` are injected as functions (not the raw client) for easy testing.
